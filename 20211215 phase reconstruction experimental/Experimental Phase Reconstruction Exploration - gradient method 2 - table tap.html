def phase_reconstruction_2(phases, phase_advance, get_deltas=False):
    results = [phases[0]]
    phase_advance %= 2*pi
    for i in range(1, len(phases)):
        expected_phase = phases[i-1] + phase_advance
        # choose between the least difference (especially when expected 
        # phase > 2pi)
        delta = (case1) if abs(case1:= phases[i]-expected_phase) < \
            abs(case2:= case1+2*pi) else case2
        results.append(delta)
    if get_deltas:
        return np.asarray(results[1:])
    results = np.cumsum(results)
    return results
